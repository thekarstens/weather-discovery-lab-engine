function computeWarningCount(pct){ var events = Array.isArray(cfg.timelineEvents) ? cfg.timelineEvents : []; var count = 0; events.forEach(function(ev){ if (!ev || !ev.displayTime) return; var startPct = displayPercentFromDisplayTime(ev.displayTime); if (!isFinite(startPct)) return; var durMin = Number(ev.activeMinutes || 30); var disp = getDisplayWindow(); var durationPct = ((durMin * 60 * 1000) / Math.max(1, (disp.end - disp.start))); var endPct = startPct + durationPct; if (pct >= startPct && pct <= endPct) count += 1; }); return count; } function updateWarningCount(pct){ var el = document.getElementById('warningCountValue'); if (!el) return; el.textContent = String(computeWarningCount(pct)); } function wireTopRowAction(mode){ applyProductMode(mode); if (mode === 'warnings') { setTicker(['Warnings view selected', 'Showing warning-focused phase at current simulator time']); } else if (mode === 'reports') { setTicker(['Storm reports view selected', 'Reports tied to current simulator time']); } else if (mode === 'radar') { setTicker(['Radar view selected', 'Use Live Doppler and storm tracker to discuss the line evolution']); } else if (mode === 'satellite') { setTicker(['Satellite view selected', 'Current-time satellite perspective for storm environment']); } else if (mode === 'temps') { setTicker(['Surface observations view selected', 'Temperature and moisture remain tied to simulator time']); } else if (mode === 'future_radar') { setTicker(['Forecast product selected', 'This is a forecast view and may extend ahead of the master clock']); } else if (mode === 'photos') { setTicker(['Storm pictures panel placeholder', 'Future upgrade: timeline-linked damage and storm photos']); } else if (mode === 'lightning') { setTicker(['Lightning view selected', 'Future upgrade: timed lightning bursts and density windows']); } }
